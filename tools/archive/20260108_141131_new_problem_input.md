<summary>–ß–µ–∫–æ—Ä 1: –û–¥—Ä–µ–¥—É–≤–∞—ö–µ —Ä–∞–≤–µ–Ω–∫–∞ –Ω–∞ –∫—Ä—É–∂–Ω–∏—Ü–∞—Ç–∞</summary>

from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # Parameters
        a = 2
        
        # Axes
        axes = Axes(
            x_range=[-1, 4, 1],
            y_range=[-4, 4, 1],
            axis_config={"color": BLACK},
        ).scale(0.8)
        
        # Fixed Elements
        line_x_a = Line(axes.c2p(a, -4), axes.c2p(a, 4), color=RED, stroke_width=3)
        circle = Circle(radius=a*0.8, color=BLACK, stroke_width=3).move_to(axes.c2p(0, a)) # radius adjusted for scale
        circle_obj = ArcByCenterCurve(axes.c2p(0, a), a*0.8, 0, TAU, color=BLACK) # Precise manim arc
        
        # Geometric Locus Curve: x = 2*a^2*y / (a^2 + y^2)
        # Using parametric representation: y is t, x is f(t)
        locus = axes.plot_parametric_curve(
            lambda t: [2 * a**2 * t / (a**2 + t**2), t, 0],
            t_range=[-4, 4],
            color=BLUE,
            stroke_width=5
        )

        # Labels (English only as per protocol)
        lbl_circle = MathTex("x^2 + (y-a)^2 = a^2", color=BLACK).scale(0.6).to_corner(UL)
        lbl_line = MathTex("x = a", color=RED).scale(0.7).next_to(line_x_a, UR, buff=0.1)
        lbl_locus = MathTex("x(a^2+y^2)=2a^2y", color=BLUE).scale(0.7).to_corner(DR)
        
        # Labeled Points
        origin = Dot(axes.c2p(0,0), color=BLACK)
        lbl_o = MathTex("O", color=BLACK).next_to(origin, DL, buff=0.1)

        # Build scene
        self.add(axes, line_x_a, circle_obj, locus, lbl_circle, lbl_line, lbl_locus, origin, lbl_o)
