from manim import *
import numpy as np

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        A = np.array([-2, -1, 0])
        B = np.array([2.5, -1, 0])
        C = np.array([1, 2.5, 0])
        
        # Helper to find foot of altitude from P to line QR
        def get_foot(P, Q, R):
            v = R - Q
            u = P - Q
            return Q + np.dot(u, v) / np.dot(v, v) * v

        A1 = get_foot(A, B, C)
        B1 = get_foot(B, A, C)
        
        # Orthocenter H is intersection of AA1 and BB1
        # Line AA1: A + t(A1-A), Line BB1: B + s(B1-B)
        # Using simple linear intersection formula
        v1 = A1 - A
        v2 = B1 - B
        # Solve A + t*v1 = B + s*v2
        # t*v1_x - s*v2_x = B_x - A_x
        # t*v1_y - s*v2_y = B_y - A_y
        mat = np.array([[v1, -v2], [v1, -v2]])
        rhs = B[:2] - A[:2]
        t, s = np.linalg.solve(mat, rhs)
        H = A + t * v1

        # Objects
        tri = Polygon(A, B, C, color=BLACK, stroke_width=4)
        alt_a = Line(A, A1, color=GRAY, stroke_style=DASHED)
        alt_b = Line(B, B1, color=GRAY, stroke_style=DASHED)
        
        dot_h = Dot(H, color=RED)
        
        # Highlights
        angle_c = Angle(Line(C, A), Line(C, B), radius=0.4, color=BLUE)
        angle_h = Angle(Line(H, B), Line(H, A), radius=0.4, color=RED)
        
        # Right angle symbols
        ra1 = RightAngle(Line(A1, A), Line(A1, C), length=0.2, color=BLACK)
        rb1 = RightAngle(Line(B1, B), Line(B1, C), length=0.2, color=BLACK)

        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, DL)
        lbl_B = MathTex("B", color=BLACK).next_to(B, DR)
        lbl_C = MathTex("C", color=BLACK).next_to(C, UP)
        lbl_H = MathTex("H", color=RED).next_to(H, DOWN, buff=0.1)
        lbl_A1 = MathTex("A_1", color=BLACK).scale(0.8).next_to(A1, UR, buff=0.05)
        lbl_B1 = MathTex("B_1", color=BLACK).scale(0.8).next_to(B1, UL, buff=0.05)
        
        gamma = MathTex("\\gamma", color=BLUE).scale(0.7).next_to(angle_c, DOWN, buff=0.1)
        result = MathTex("\\angle AHB = 180^\\circ - \\gamma", color=BLACK).to_edge(DOWN)

        # Build final frame
        self.add(tri, alt_a, alt_b, dot_h, ra1, rb1, angle_c, angle_h)
        self.add(lbl_A, lbl_B, lbl_C, lbl_H, lbl_A1, lbl_B1, gamma, result)
