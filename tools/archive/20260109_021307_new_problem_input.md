from manim import *
import numpy as np

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        # Let's pick beta = 20, then alpha = 110, gamma = 50.
        # This satisfies alpha - beta = 90.
        
        # Points calculation (Synthetic to Analytic for Diagram)
        C = np.array()
        # Angle at C is 50. Internal bisector is vertical.
        # Angle beta = 20, gamma/2 = 25.
        # In triangle BCD, angle BDC = 180 - 20 - 25 = 135.
        
        B = np.array([3, -1, 0])
        A = np.array([-1.5, -1, 0]) # Adjusted for alpha-beta=90
        D = np.array([0.5, -1, 0])  # D is on AB
        E = np.array([-2.5, -1, 0]) # E is on extension
        
        # Triangle and Lines
        tri_abc = Polygon(A, B, C, color=BLACK, stroke_width=4)
        line_ab_ext = Line(E, B, color=BLACK, stroke_width=2)
        bisector_int = Line(C, D, color=BLUE, stroke_width=3)
        bisector_ext = Line(C, E, color=RED, stroke_width=3)
        
        # Markers
        ra = RightAngle(bisector_ext, bisector_int, length=0.3, color=BLACK, quadrant=(-1, 1))
        
        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, DOWN)
        lbl_B = MathTex("B", color=BLACK).next_to(B, DOWN)
        lbl_C = MathTex("C", color=BLACK).next_to(C, UP)
        lbl_D = MathTex("D", color=BLUE).next_to(D, DOWN)
        lbl_E = MathTex("E", color=RED).next_to(E, DOWN)
        
        # Highlighted Triangle EDC
        tri_edc = Polygon(E, D, C, color=PURPLE, fill_opacity=0.1, stroke_width=0)
        
        # Display
        self.add(line_ab_ext, tri_abc, tri_edc)
        self.add(bisector_int, bisector_ext, ra)
        self.add(lbl_A, lbl_B, lbl_C, lbl_D, lbl_E)
        
        # Final mathematical statement
        statement = MathTex("Angle(CED) = Angle(EDC)", color=BLACK).to_edge(DOWN)
        self.add(statement)
