from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        # Angles in degrees: 36, 36, 108
        alpha = 36 * DEGREES
        base_len = 5
        side_len = base_len / (2 * np.cos(alpha))
        height = (base_len / 2) * np.tan(alpha)
        
        # Vertices
        D = ORIGIN
        A = LEFT * (base_len / 2)
        B = RIGHT * (base_len / 2)
        C = UP * height
        E = DOWN * height
        
        # Triangle and Rhombus
        tri = Polygon(A, B, C, color=BLACK, stroke_width=4)
        rhombus = Polygon(A, C, B, E, color=GRAY, stroke_width=2, stroke_opacity=0.5)
        
        # Height and Bisector
        height_line = Line(C, D, color=RED, stroke_width=4)
        # Bisector AL calculation
        # L is on BC. BC line: y = height - (height/ (base_len/2)) * (x - 0) ? No.
        # Vector BC = C - B. Point L = A + t * (Bisector Vector)
        bisector_vec = rotate_vector(RIGHT, alpha / 2)
        # Intersection of AL and BC
        # AL: A + t*bisector_vec
        # BC: B + u*(C-B)
        # A + t*v = B + u*w
        v = bisector_vec
        w = C - B
        mat = np.array([[v, -w], [v, -w]])
        rhs = B - A
        t, u = np.linalg.solve(mat[:2, :2], rhs[:2])
        L = A + t * v
        
        bisector_line = Line(A, L, color=BLUE, stroke_width=4)
        
        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, LEFT)
        lbl_B = MathTex("B", color=BLACK).next_to(B, RIGHT)
        lbl_C = MathTex("C", color=BLACK).next_to(C, UP)
        lbl_L = MathTex("L", color=BLACK).next_to(L, UR, buff=0.1)
        lbl_D = MathTex("D", color=BLACK).next_to(D, DL, buff=0.1)
        
        # Final result text
        res = MathTex("36^\\circ, 36^\\circ, 108^\\circ", color=BLACK).to_edge(DOWN)

        # Draw
        self.add(rhombus, tri, height_line, bisector_line)
        self.add(lbl_A, lbl_B, lbl_C, lbl_L, lbl_D, res)
