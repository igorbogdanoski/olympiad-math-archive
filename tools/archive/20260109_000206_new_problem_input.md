```python
from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        # Vertices for the quadrilateral ABCD
        # C is origin, B on x-axis, D on y-axis
        C = ORIGIN
        B = RIGHT * 3
        D = UP * 4
        
        # Point P on CD
        P = UP * 1.5
        
        # A calculated to satisfy angles (simplified for visualization)
        A = np.array([2.5, 4.5, 0])
        
        # Reflection of A across CD (y-axis)
        A_prime = np.array([-2.5, 4.5, 0])
        
        # Drawing
        quad = Polygon(A, B, C, D, color=BLACK, stroke_width=4)
        line_ap = Line(A, P, color=BLACK)
        line_bp = Line(B, P, color=BLACK)
        line_pa_prime = Line(P, A_prime, color=BLUE, stroke_style=DASHED)
        
        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, UR)
        lbl_B = MathTex("B", color=BLACK).next_to(B, RIGHT)
        lbl_C = MathTex("C", color=BLACK).next_to(C, DL)
        lbl_D = MathTex("D", color=BLACK).next_to(D, UP)
        lbl_P = MathTex("P", color=BLACK).next_to(P, LEFT)
        lbl_A_prime = MathTex("A'", color=BLUE).next_to(A_prime, UL)
        
        # Highlights
        angle_alpha1 = Arc(radius=0.4, start_angle=PI/2, angle=0.5, color=RED).move_to(P)
        angle_alpha2 = Arc(radius=0.4, start_angle=-PI/2, angle=-0.5, color=RED).move_to(P)
        
        # Logic Text
        logic = MathTex("AP + BP = A'B", color=BLACK).to_corner(UL)
        final = MathTex("BC = \\frac{A'B}{2}", color=RED).next_to(logic, DOWN)

        # Build Frame
        self.add(quad, line_ap, line_bp, line_pa_prime)
        self.add(lbl_A, lbl_B, lbl_C, lbl_D, lbl_P, lbl_A_prime)
        self.add(logic, final)
```
