from manim import *
import numpy as np

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        # Vertices for triangle ABC where AB < AC
        A = np.array([-0.5, 2.0, 0])
        B = np.array([-2.5, -1.0, 0])
        C = np.array([3.0, -1.0, 0])
        
        # Foot of altitude D
        # D is projection of A onto BC (which is y = -1)
        D = np.array([-0.5, -1.0, 0])
        
        # Intersection of Bisector E on BC
        # Using Angle Bisector Theorem or vector rotation
        def get_bisector_point(p_a, p_b, p_c):
            v_ab = p_b - p_a
            v_ac = p_c - p_a
            unit_ab = v_ab / np.linalg.norm(v_ab)
            unit_ac = v_ac / np.linalg.norm(v_ac)
            bisector_dir = unit_ab + unit_ac
            # Line A + t * bisector_dir intersects BC (y = -1)
            # A_y + t * dir_y = -1 => t = (-1 - A_y) / dir_y
            t = (-1.0 - p_a) / bisector_dir
            return p_a + t * bisector_dir

        E = get_bisector_point(A, B, C)
        
        # Geometry Objects
        tri = Polygon(A, B, C, color=BLACK, stroke_width=4)
        altitude = Line(A, D, color=GRAY, stroke_style=DASHED)
        bisector = Line(A, E, color=BLUE, stroke_width=3)
        
        # Highlighting angle DAE
        angle_dae = Angle(Line(A, E), Line(A, D), radius=0.5, color=RED)
        
        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, UP)
        lbl_B = MathTex("B", color=BLACK).next_to(B, DL)
        lbl_C = MathTex("C", color=BLACK).next_to(C, DR)
        lbl_D = MathTex("D", color=BLACK).next_to(D, DOWN)
        lbl_E = MathTex("E", color=BLUE).next_to(E, DOWN)
        
        # Math result
        formula = MathTex("\\angle DAE = \\frac{\\beta - \\gamma}{2}", color=BLACK).to_edge(DOWN)

        # Final Render State
        self.add(tri, altitude, bisector, angle_dae)
        self.add(lbl_A, lbl_B, lbl_C, lbl_D, lbl_E, formula)
