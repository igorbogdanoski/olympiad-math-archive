```python
from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION & GEOMETRY ---
        # Radii of small semicircles
        r1 = 1.25
        r2 = 1.75
        R_large = r1 + r2 # Should be 3.0
        
        # Positions on the X-axis
        # R is at -3, S is at 3, T is the junction point
        point_R = LEFT * R_large
        point_S = RIGHT * R_large
        point_T = point_R + RIGHT * (2 * r1)
        
        # Centers of the three semicircles
        center_large = ORIGIN
        center_1 = point_R + RIGHT * r1
        center_2 = point_S + LEFT * r2
        
        # --- OBJECTS ---
        # Creating the shapes as Sectors (semicircles) for proper filling
        large_semi = Sector(
            outer_radius=R_large, 
            angle=PI, 
            start_angle=0, 
            color=BLACK, 
            fill_opacity=0, 
            stroke_width=4
        ).shift(DOWN * 0.5)
        
        small_semi1 = Sector(
            outer_radius=r1, 
            angle=PI, 
            start_angle=0, 
            color=BLACK, 
            fill_opacity=0, 
            stroke_width=4
        ).move_to(center_1 + DOWN * 0.5, aligned_edge=DOWN)
        
        small_semi2 = Sector(
            outer_radius=r2, 
            angle=PI, 
            start_angle=0, 
            color=BLACK, 
            fill_opacity=0, 
            stroke_width=4
        ).move_to(center_2 + DOWN * 0.5, aligned_edge=DOWN)

        # --- SHADED AREA (ARBELOS) ---
        # We create the blue area by subtracting the two small ones from the large one
        # Note: In Manim Community, we can use Difference for VMobjects
        large_filled = Sector(outer_radius=R_large, angle=PI, fill_color=BLUE, fill_opacity=0.3, stroke_width=0).shift(DOWN*0.5)
        small1_filled = Sector(outer_radius=r1, angle=PI, fill_color=WHITE, fill_opacity=1.0, stroke_width=0).move_to(center_1 + DOWN*0.5, aligned_edge=DOWN)
        small2_filled = Sector(outer_radius=r2, angle=PI, fill_color=WHITE, fill_opacity=1.0, stroke_width=0).move_to(center_2 + DOWN*0.5, aligned_edge=DOWN)
        
        # Altitude AT
        h_at = ( (2*r1) * (2*r2) )**0.5 / 1 # AT^2 = RT * TS
        point_A = point_T + UP * h_at
        line_at = Line(point_T + DOWN*0.5, point_A + DOWN*0.5, color=RED, stroke_width=5)
        
        # Main diameter line
        base_line = Line(point_R + DOWN*0.5, point_S + DOWN*0.5, color=BLACK, stroke_width=4)

        # --- LABELS ---
        labels = VGroup(
            MathTex("R", color=BLACK).next_to(point_R + DOWN*0.5, DOWN),
            MathTex("S", color=BLACK).next_to(point_S + DOWN*0.5, DOWN),
            MathTex("T", color=BLACK).next_to(point_T + DOWN*0.5, DOWN),
            MathTex("A", color=BLACK).next_to(point_A + DOWN*0.5, UP),
            MathTex("Area Ratio = 2:1", color=BLUE).to_corner(UR)
        )

        # Rendering order: Fill first, then outlines, then labels
        self.add(large_filled, small1_filled, small2_filled)
        self.add(large_semi, small_semi1, small_semi2, base_line, line_at, labels)
```
