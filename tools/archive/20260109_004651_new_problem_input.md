from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # Define vertices of triangle ABC
        A = np.array([-2, -1, 0])
        B = np.array([2, -1, 0])
        C = np.array()
        
        # Incenter Calculation
        # Sides lengths: a = BC, b = AC, c = AB
        c = np.linalg.norm(B - A)
        b = np.linalg.norm(C - A)
        a = np.linalg.norm(C - B)
        I = (a * A + b * B + c * C) / (a + b + c)
        
        # Create shapes
        triangle = Polygon(A, B, C, color=BLACK, stroke_width=4)
        bisector_a = Line(A, I, color=GRAY, stroke_width=2, stroke_opacity=0.6)
        bisector_b = Line(B, I, color=GRAY, stroke_width=2, stroke_opacity=0.6)
        inner_tri = Polygon(A, B, I, color=BLUE, fill_opacity=0.1, stroke_width=3)
        
        # Dots and Labels
        dot_i = Dot(I, color=RED)
        lbl_A = MathTex("A", color=BLACK).next_to(A, DL)
        lbl_B = MathTex("B", color=BLACK).next_to(B, DR)
        lbl_C = MathTex("C", color=BLACK).next_to(C, UP)
        lbl_I = MathTex("I", color=RED).next_to(I, UP, buff=0.1)
        
        # Angle markers
        # Using simple MathTex since we can't use Cyrillic
        alpha_2 = MathTex("\\alpha/2", color=BLACK).scale(0.6).move_to(A + [0.6, 0.3, 0])
        beta_2 = MathTex("\\beta/2", color=BLACK).scale(0.6).move_to(B + [-0.6, 0.3, 0])
        formula = MathTex("\\angle AIB = 90^\\circ + \\frac{\\gamma}{2}", color=BLACK).to_edge(DOWN)
        
        # Add everything to scene
        self.add(triangle, bisector_a, bisector_b, inner_tri, dot_i)
        self.add(lbl_A, lbl_B, lbl_C, lbl_I, alpha_2, beta_2, formula)
