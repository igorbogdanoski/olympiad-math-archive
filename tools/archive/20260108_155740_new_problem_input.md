from manim import *
import numpy as np

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        # Fixed the V = np.array() error by providing valid coordinates 
        # Defining a valid tangential quadrilateral
        V = np.array()
        r_val = 1.4
        circle = Circle(radius=r_val, color=BLUE, stroke_width=2).move_to(V)
        
        # Approximate tangent points to construct vertices
        A = np.array([-2.5, 2.0, 0])
        B = np.array([-2.0, -2.5, 0])
        C = np.array([3.5, -1.5, 0])
        D = np.array([1.5, 2.5, 0])
        
        quad = Polygon(A, B, C, D, color=BLACK, stroke_width=4)
        
        # Midpoints
        M = (B + C) / 2
        N = (A + D) / 2
        
        # Midpoint line MN
        line_mn = Line(M, N, color=RED, stroke_width=4)
        
        # Labels (Strictly English/Math Symbols)
        lbl_a = MathTex("A", color=BLACK).next_to(A, UL)
        lbl_b = MathTex("B", color=BLACK).next_to(B, DL)
        lbl_c = MathTex("C", color=BLACK).next_to(C, DR)
        lbl_d = MathTex("D", color=BLACK).next_to(D, UR)
        lbl_m = MathTex("M", color=BLACK).next_to(M, DOWN)
        lbl_n = MathTex("N", color=BLACK).next_to(N, UP)
        lbl_r = MathTex("r", color=BLUE).move_to(V + UP*0.4 + RIGHT*0.4)
        
        # Radius line for visualization
        rad_line = Line(V, V + r_val * np.array([np.cos(PI/4), np.sin(PI/4), 0]), color=BLUE)

        # Final Inequality Result
        result = MathTex("MN \geq 2r", color=BLACK).to_edge(DOWN, buff=1)

        # Rendering
        self.add(quad, circle, line_mn, rad_line)
        self.add(lbl_a, lbl_b, lbl_c, lbl_d, lbl_m, lbl_n, lbl_r, result)
