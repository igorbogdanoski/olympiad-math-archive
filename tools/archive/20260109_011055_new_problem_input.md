        v_bd = D - B
        dir_b = v_ba/np.linalg.norm(v_ba) + v_bd/np.linalg.norm(v_bd)
        
        # Find intersection Y (A + t*dir_a = B + s*dir_b)
        m = np.array([dir_a[:2], -dir_b[:2]]).T
        rhs = (B - A)[:2]
        sol = np.linalg.solve(m, rhs)
        Y = A + sol * dir_a
        
        # --- DRAWING ---
        triangle = Polygon(A, B, C, color=BLACK, stroke_width=4)
        ext_line = Line(C, D, color=GRAY, stroke_style=DASHED)
        
        bisector_a = Line(A, Y, color=BLUE, stroke_width=3)
        bisector_b = Line(B, Y, color=BLUE, stroke_width=3)
        
        # Highlights
        dot_y = Dot(Y, color=RED)
        angle_indicator = Angle(Line(Y, A), Line(Y, B), radius=0.5, color=RED)
        
        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, UP)
        lbl_B = MathTex("B", color=BLACK).next_to(B, DOWN)
        lbl_C = MathTex("C", color=BLACK).next_to(C, DR)
        lbl_Y = MathTex("Y", color=RED).next_to(Y, LEFT)
        
        gamma_half = MathTex("\\gamma / 2", color=BLACK).next_to(angle_indicator, DR, buff=0.1)
        res = MathTex("\\angle AYB = \\gamma / 2", color=BLACK).to_edge(DOWN)
        
        # Adding to scene
        self.add(ext_line, triangle, bisector_a, bisector_b, dot_y, angle_indicator)
        self.add(lbl_A, lbl_B, lbl_C, lbl_Y, gamma_half, res)
