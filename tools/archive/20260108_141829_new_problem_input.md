from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # Radii calculations for drawing
        r_small = 1.0
        phi_sq = (3 + 5**0.5) / 2
        R_large = r_small * phi_sq
        L = 3 * R_large - r_small
        H = 2 * R_large
        
        # Scaling to fit screen
        scale_factor = 0.6
        L_s, H_s, R_s, r_s = L*scale_factor, H*scale_factor, R_large*scale_factor, r_small*scale_factor
        
        # Rectangle
        rect = Rectangle(width=L_s, height=H_s, color=BLACK, stroke_width=4).shift(LEFT*L_s/2 + DOWN*H_s/2)
        # Shift to place P at origin-ish
        rect.move_to(ORIGIN)
        
        # Vertices
        P = rect.get_corner(DL)
        Q = rect.get_corner(DR)
        S = rect.get_corner(UR)
        T = rect.get_corner(UL)
        
        # Circles
        O1 = P + RIGHT*R_s + UP*R_s
        O2 = Q + LEFT*r_s + UP*r_s
        k1 = Circle(radius=R_s, color=BLACK, stroke_width=3).move_to(O1)
        k2 = Circle(radius=r_s, color=BLACK, stroke_width=3).move_to(O2)
        
        # Tangent Line CS
        # Point C is on line O1O2 at distance R/(R+r) from O1
        vec_O1O2 = O2 - O1
        C = O1 + vec_O1O2 * (R_large / (R_large + r_small))
        line_cs = Line(C, S, color=RED, stroke_width=4)
        
        # Labels
        lbl_p = MathTex("P", color=BLACK).next_to(P, DL)
        lbl_q = MathTex("Q", color=BLACK).next_to(Q, DR)
        lbl_s = MathTex("S", color=BLACK).next_to(S, UR)
        lbl_t = MathTex("T", color=BLACK).next_to(T, UL)
        lbl_r_cap = MathTex("R", color=BLACK).move_to(O1 + UP*0.3)
        lbl_r_low = MathTex("r", color=BLACK).move_to(O2)
        
        # Construction highlights
        dot_c = Dot(C, color=RED)
        lbl_c = MathTex("C", color=RED).next_to(C, DOWN, buff=0.1)

        final_res = MathTex("R = r \\cdot \\phi^2", color=BLUE).to_corner(UL)

        self.add(rect, k1, k2, line_cs, dot_c, lbl_c)
        self.add(lbl_p, lbl_q, lbl_s, lbl_t, lbl_r_cap, lbl_r_low, final_res)
