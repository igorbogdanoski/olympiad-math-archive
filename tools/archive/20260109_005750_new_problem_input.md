from manim import *
import numpy as np

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION (Points) ---
        # Vertices for an acute triangle
        A = np.array([-2.5, -1.0, 0])
        B = np.array([2.5, -1.0, 0])
        C = np.array([0.5, 2.5, 0])
        
        # Helper function for altitude foot
        def get_foot(P, Q, R):
            v = R - Q
            u = P - Q
            foot = Q + np.dot(u, v) / np.dot(v, v) * v
            return foot

        A1 = get_foot(A, B, C)
        B1 = get_foot(B, A, C)
        
        # Intersection calculation (Orthocenter H) - FIXED
        def intersect_lines(P1, d1, P2, d2):
            # P1 + t*d1 = P2 + s*d2, solve for t,s in 2D
            A_mat = np.array([d1[:2], -d2[:2]]).T
            b = P2[:2] - P1[:2]
            t, s = np.linalg.solve(A_mat, b)
            return P1 + t * d1

        H = intersect_lines(A, A1 - A, B, B1 - B)

        # --- DRAWING ---
        triangle = Polygon(A, B, C, color=BLACK, stroke_width=4)
        alt_a = Line(A, A1, color=GRAY, stroke_style=DASHED)
        alt_b = Line(B, B1, color=GRAY, stroke_style=DASHED)
        
        # Right angles
        ra1 = RightAngle(Line(A1, A), Line(A1, C), length=0.2, color=BLACK)
        rb1 = RightAngle(Line(B1, B), Line(B1, C), length=0.2, color=BLACK)
        
        # Points
        dot_h = Dot(H, color=RED)
        
        # Angle arcs
        angle_c = Angle(Line(C, A), Line(C, B), radius=0.4, color=BLUE)
        angle_h = Angle(Line(H, B), Line(H, A), radius=0.4, color=RED)

        # Labels (Strictly English/Math)
        lbl_A = MathTex("A", color=BLACK).next_to(A, DL)
        lbl_B = MathTex("B", color=BLACK).next_to(B, DR)
        lbl_C = MathTex("C", color=BLACK).next_to(C, UP)
        lbl_H = MathTex("H", color=RED).next_to(H, DOWN, buff=0.1)
        lbl_A1 = MathTex("A_1", color=BLACK).scale(0.8).next_to(A1, UR, buff=0.05)
        lbl_B1 = MathTex("B_1", color=BLACK).scale(0.8).next_to(B1, UL, buff=0.05)
        
        gamma = MathTex("\\gamma", color=BLUE).scale(0.8).next_to(angle_c, DOWN, buff=0.1)
        res = MathTex("\\angle AHB = 180^\\circ - \\gamma", color=BLACK).to_edge(DOWN)

        # Final Render State
        self.add(triangle, alt_a, alt_b, ra1, rb1, dot_h, angle_c, angle_h)
        self.add(lbl_A, lbl_B, lbl_C, lbl_H, lbl_A1, lbl_B1, gamma, res)
