        O = ORIGIN
        O1 = np.array([-r1, 0, 0])
        A = np.array([-r, 0, 0])
        
        # Circles
        k = Circle(radius=r, color=BLACK, stroke_width=4).move_to(O)
        k1 = Circle(radius=r1, color=BLACK, stroke_width=3).move_to(O1)
        
        # Chord AB
        angle = 40 * DEGREES
        B = np.array([r * np.cos(angle), r * np.sin(angle), 0])
        chord = Line(A, B, color=RED, stroke_width=4)
        
        # Intersection P with k1
        # P is the point on k1 such that A, P, B are collinear.
        # Since AO is diameter, angle AP O is 90 if P is on diameter? No.
        # In similarity, P is at distance half of AB from A.
        P = A + 0.5 * (B - A)
        
        # Triangle and Segments logic
        tri_aob = Polygon(A, O, B, color=BLACK, fill_opacity=0.1, fill_color=GRAY)
        tri_ao1p = Polygon(A, O1, P, color=BLACK, fill_opacity=0.2, fill_color=BLUE)
        
        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, LEFT)
        lbl_O = MathTex("O", color=BLACK).next_to(O, DOWN)
        lbl_O1 = MathTex("O_1", color=BLACK).next_to(O1, DOWN)
        lbl_B = MathTex("B", color=BLACK).next_to(B, UR)
        lbl_P = MathTex("P", color=BLACK).next_to(P, UP)
        
        # Sector/Segment Highlights
        # (Simplified as arcs for visual representation)
        arc_k = Arc(radius=r, start_angle=PI, angle=angle, color=RED)
        arc_k1 = Arc(radius=r1, start_angle=PI, angle=angle, color=BLUE)

        # Result Text
        result = MathTex("S_2 = 4 S_1", color=BLACK).to_edge(DOWN)

        # Drawing
        self.add(k, k1, chord, tri_aob, tri_ao1p)
        self.add(lbl_A, lbl_O, lbl_O1, lbl_B, lbl_P, result)
