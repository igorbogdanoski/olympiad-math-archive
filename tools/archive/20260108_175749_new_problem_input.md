from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        # Vertices of ABC (Equilateral)
        side = 5
        A = np.array([-side/2, 0, 0])
        B = np.array([side/2, 0, 0])
        C = np.array([0, side * (3**0.5)/2, 0])
        
        tri_abc = Polygon(A, B, C, color=BLACK, stroke_width=4)
        
        # Points E and F (EF || BC is not possible if ABC is equilateral and A is at bottom)
        # Re-orienting: C as top vertex, AB as base.
        # Let's use standard coordinates.
        A_p = np.array([-2, -1, 0])
        B_p = np.array([2, -1, 0])
        C_p = np.array([0, 2.46, 0]) # Equilateral height
        
        # Point E on AC, F on AB
        ratio = 0.4
        E = A_p + (C_p - A_p) * ratio
        # To make EF || BC, F must be on AB? No, problem says E on AC, F on AB.
        # Actually, for EF || BC, F must be on AB.
        # Let's fix triangle orientation: A at top.
        A = np.array([0, 2.5, 0])
        B = np.array([-2.2, -1.3, 0])
        C = np.array([2.2, -1.3, 0])
        
        ratio = 0.6
        E = A + (C - A) * ratio
        F = A + (B - A) * ratio # EF is parallel to BC
        
        # Centroid G of AFE
        G = (A + F + E) / 3
        
        # Midpoint M of BE
        M = (B + E) / 2
        
        # Visuals
        main_tri = Polygon(A, B, C, color=BLACK)
        small_tri = Polygon(A, F, E, color=BLUE, stroke_width=2)
        line_be = Line(B, E, color=RED)
        dot_g = Dot(G, color=BLACK)
        dot_m = Dot(M, color=RED)
        line_gc = Line(G, C, color=BLACK, stroke_width=2)
        line_gm = Line(G, M, color=BLACK, stroke_width=2)
        
        # Labels
        lbl_a = MathTex("A", color=BLACK).next_to(A, UP)

        self.add(main_tri, small_tri, line_be, dot_g, dot_m, line_gc, line_gm)
        self.add(lbl_a, lbl_b, lbl_c, lbl_g, lbl_m, lbl_60)
