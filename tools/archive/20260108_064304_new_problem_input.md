from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # Triangle vertices
        # Height of triangle with sides 3, 3, 4:
        # h^2 = 3^2 - 2^2 = 5 => h = sqrt(5) approx 2.23
        A = UP * 2.23 + LEFT * 0
        B = LEFT * 2
        C = RIGHT * 2
        
        triangle = Polygon(A, B, C, color=BLACK, stroke_width=4)
        
        # Points D and E
        # DC = 3 (from C towards B), so D is at C + 3*LEFT = 2 - 3 = -1
        D = LEFT * 1
        # EC = 2 (from C towards A)
        # Vector CA = A - C = (0, 2.23) - (2, 0) = (-2, 2.23)
        # Unit vector CA = (-2, 2.23) / 3
        # E = C + (2/3) * (A - C)
        E = C + (2.0/3.0) * (A - C)
        
        line_de = Line(D, E, color=RED, stroke_width=4)
        
        # Labels
        labels = VGroup(
            MathTex("A", color=BLACK).next_to(A, UP),
            MathTex("B", color=BLACK).next_to(B, DOWN),
            MathTex("C", color=BLACK).next_to(C, DOWN),
            MathTex("D", color=BLACK).next_to(D, DOWN),
            MathTex("E", color=BLACK).next_to(E, RIGHT, buff=0.1),
            MathTex("3", color=BLACK).move_to((A + B)/2 + LEFT*0.3),
            MathTex("4", color=BLACK).next_to((B + C)/2, DOWN, buff=0.5)
        )
        
        # Highlight segments on base
        seg_bd = Line(B, D, color=BLUE, stroke_width=6)
        seg_dc = Line(D, C, color=GREEN, stroke_width=6)
        
        label_bd = MathTex("1", color=BLUE).next_to(seg_bd, UP, buff=0.1)
        label_dc = MathTex("3", color=GREEN).next_to(seg_dc, UP, buff=0.1)

        self.add(triangle, line_de, labels, seg_bd, seg_dc, label_bd, label_dc)
