        t_afe = Polygon(A, F, E, color=BLUE, fill_opacity=0.1, stroke_width=3)
        l_F = MathTex("F", color=BLUE).next_to(F, LEFT)
        l_E = MathTex("E", color=BLUE).next_to(E, RIGHT)
        
        self.play(FadeIn(t_afe), Write(l_F), Write(l_E))
        
        # Centroid G of AFE
        G = (A + F + E) / 3
        d_G = Dot(G, color=RED)
        l_G = MathTex("G", color=RED).next_to(G, UP, buff=0.1)
        
        self.play(Create(d_G), Write(l_G))
        
        # Line BE
        line_BE = Line(B, E, color=GRAY)
        self.play(Create(line_BE))
        
        # Midpoint M of BE
        M = (B + E) / 2
        d_M = Dot(M, color=ORANGE)
        l_M = MathTex("M", color=ORANGE).next_to(M, DOWN, buff=0.1)
        
        self.play(Create(d_M), Write(l_M))
        
        # Connect G, M, C
        line_GC = Line(G, C, color=GREEN)
        line_GM = Line(G, M, color=GREEN)
        line_MC = Line(M, C, color=GREEN) # Just to show the triangle, not strictly needed for angle
        
        self.play(Create(line_GC), Create(line_GM))
        
        # Angle MGC
        # Ideally we visualize the rotation
        
        # Rotation Animation
        # Copy AFE and rotate around C by -60 degrees (CW) to match A -> B orientation check
        # Wait, A is (0, sqrt3), B is (-1,0), C is (1,0).
        # Vector CA = (-1, sqrt3). Vector CB = (-2, 0). 
        # Angle CA is 120 deg. Angle CB is 180 deg. Diff is 60 deg CCW.
        # So Rotation of A around C by +60 deg lands on B? No, 120 to 180 is +60. Yes.
        # So we rotate +60 degrees (Counter-Clockwise).
        
        g_prime_pos = C + (np.array([[np.cos(np.pi/3), -np.sin(np.pi/3), 0], 
                                     [np.sin(np.pi/3), np.cos(np.pi/3), 0], 
                                     [0,0,1]]) @ (G - C))
        
        d_G_prime = Dot(g_prime_pos, color=PURPLE)
        l_Gp = MathTex("G'", color=PURPLE).next_to(g_prime_pos, RIGHT)
        
        # Visualizing the parallelogram
        para = Polygon(B, g_prime_pos, E, G, color=PURPLE, stroke_opacity=0.5, stroke_dash_pattern=[5,5])
        
        self.play(
            Rotate(d_G.copy(), angle=np.pi/3, about_point=C, rate_func=smooth),
            run_time=2
        )
        self.add(d_G_prime, l_Gp)
        self.wait(1)
        
        # Show G-G'-C is equilateral
        tri_rot = Polygon(G, g_prime_pos, C, color=GREEN, fill_opacity=0.1)
        self.play(FadeIn(tri_rot))
        
        # Show Parallelogram
        self.play(Create(para))
        
        # Emphasize M is intersection
        self.play(Indicate(d_M, scale_factor=1.5, color=ORANGE))
        
        # Final Text
        result_text = MathTex(r"\angle MGC = 60^\circ", color=BLACK).to_edge(UP)
        self.play(Write(result_text))
        
        self.wait(3)
