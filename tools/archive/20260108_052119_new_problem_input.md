from manim import *

class DigitsPuzzle(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # Title
        title = Text("–ó–∞–¥–∞—á–∞ 1: –ò–≥—Ä–∞ —Å–æ —Ü–∏—Ñ—Ä–∏", font_size=36, color=BLACK).to_edge(UP)
        self.play(Write(title))
        
        # Legend
        odd_sample = Square(side_length=0.5, color=BLUE, fill_opacity=0.5).to_edge(LEFT).shift(UP)
        odd_text = Text("–ù–µ–ø–∞—Ä–Ω–∞ (x)", font_size=24, color=BLACK).next_to(odd_sample, RIGHT)
        even_sample = Square(side_length=0.5, color=RED, fill_opacity=0.5).next_to(odd_sample, DOWN, buff=0.5)
        even_text = Text("–ü–∞—Ä–Ω–∞ (y)", font_size=24, color=BLACK).next_to(even_sample, RIGHT)
        
        self.play(FadeIn(odd_sample), Write(odd_text), FadeIn(even_sample), Write(even_text))
        
        # Step 1: Monday
        # Let's simulate a small number: Odd, Even, Odd (x=2, y=1)
        # Total L1 = 3
        
        mon_group = VGroup(
            Square(side_length=0.8, color=BLUE, fill_opacity=0.5), # Odd
            Square(side_length=0.8, color=RED, fill_opacity=0.5),  # Even
            Square(side_length=0.8, color=BLUE, fill_opacity=0.5)  # Odd
        ).arrange(RIGHT, buff=0.2).shift(UP*0.5)
        
        mon_label = Text("–ü–æ–Ω–µ–¥–µ–ª–Ω–∏–∫ (L1)", font_size=28, color=BLACK).next_to(mon_group, LEFT, buff=1)
        
        self.play(Write(mon_label), Create(mon_group))
        self.wait(1)
        
        # Step 2: Tuesday
        # Add '2' (Even/Red) next to every Odd/Blue
        
        tue_group = VGroup()
        # Reconstruct the group with insertions
        # O -> O E
        # E -> E
        # O -> O E
        
        sq1 = mon_group[0].copy()
        new2_1 = Square(side_length=0.8, color=RED, fill_opacity=0.8).next_to(sq1, RIGHT, buff=0.1) # The added '2'
        text2_1 = MathTex("2", color=WHITE).move_to(new2_1.get_center())
        
        sq2 = mon_group[1].copy().next_to(new2_1, RIGHT, buff=0.2)
        
        sq3 = mon_group[2].copy().next_to(sq2, RIGHT, buff=0.2)
        new2_2 = Square(side_length=0.8, color=RED, fill_opacity=0.8).next_to(sq3, RIGHT, buff=0.1) # The added '2'
        text2_2 = MathTex("2", color=WHITE).move_to(new2_2.get_center())
        
        tue_visual = VGroup(sq1, new2_1, text2_1, sq2, sq3, new2_2, text2_2).move_to(ORIGIN).shift(DOWN*0.5)
        tue_label = Text("–í—Ç–æ—Ä–Ω–∏–∫ (L2)", font_size=28, color=BLACK).next_to(tue_visual, LEFT, buff=1).align_to(mon_label, LEFT)
        
        self.play(
            TransformFromCopy(mon_group, tue_visual),
            Write(tue_label)
        )
        self.wait(1)
        
        # Highlight Evens on Tuesday
        # Original Even (1) + Added '2's (2) = 3 Evens. Total length Monday was 3.
        # This is the key insight.
        
        brace = Brace(tue_visual, DOWN, color=BLACK)
        explanation = MathTex("L_2 = L_1 + x", color=BLACK).next_to(brace, DOWN)
        
        self.play(Create(brace), Write(explanation))
        self.wait(1)
        
        # Step 3: Wednesday logic (Text only to save space)
        # "Wednesday adds digits equal to count of Evens on Tuesday"
        
        logic_text = VGroup(
            Text("–ü–∞—Ä–Ω–∏ –≤–æ –í—Ç–æ—Ä–Ω–∏–∫ = (–û—Ä–∏–≥–∏–Ω–∞–ª–Ω–∏ –ü–∞—Ä–Ω–∏) + (–ù–æ–≤–∏ '2')", font_size=24, color=BLACK),
            MathTex("= y + x = L_1", color=RED)
        ).arrange(DOWN).next_to(explanation, DOWN, buff=0.5)
        
        self.play(Write(logic_text))
        self.wait(2)
        
        # Final Equation
        final_eq = MathTex("L_3 - L_1 = (L_2 + L_1) - L_1 = L_2", color=BLUE).scale(1.2).to_edge(DOWN)
        final_res = Text("–ó–Ω–∞—á–∏ L2 = 2025", font_size=32, color=BLUE).next_to(final_eq, DOWN)
        
        self.play(Write(final_eq))
        self.wait(2)
