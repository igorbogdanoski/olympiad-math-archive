from manim import *

class VectorTriangle(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # Coordinates
        # Let Q be at (0, 3), A at (-2, 0), B at (2, 0)
        # Then T is midpoint of QA: (-1, 1.5)
        # S is midpoint of QB: (1, 1.5)
        # C is centroid of QAB: (0, 1)
        
        Q = np.array([0, 3, 0])
        A = np.array([-3, -1, 0])
        B = np.array([3, -1, 0])
        
        # Midpoints
        T = (Q + A) / 2
        S = (Q + B) / 2
        R = (A + B) / 2 # Midpoint of AB
        
        # Centroid C
        C = (Q + A + B) / 3
        
        # Vectors/Lines
        triangle = Polygon(A, B, Q, color=BLACK)
        median_AS = Line(A, S, color=BLUE)
        median_BT = Line(B, T, color=BLUE)
        median_QR = Line(Q, R, color=GREEN) # The line containing C
        
        # Labels
        label_Q = MathTex("Q", color=BLACK).next_to(Q, UP)
        label_A = MathTex("A", color=BLACK).next_to(A, DL)
        label_B = MathTex("B", color=BLACK).next_to(B, DR)
        label_T = MathTex("T", color=BLACK).next_to(T, LEFT)
        label_S = MathTex("S", color=BLACK).next_to(S, RIGHT)
        label_C = MathTex("C", color=RED).next_to(C, DR, buff=0.1)
        label_R = MathTex("R", color=BLACK).next_to(R, DOWN)
        
        # Animation Sequence
        self.play(Create(triangle), Write(label_Q), Write(label_A), Write(label_B))
        self.wait(0.5)
        
        self.play(Create(median_AS), Write(label_S))
        self.play(Create(median_BT), Write(label_T))
        
        # Intersection C
        dot_C = Dot(C, color=RED)
        self.play(FadeIn(dot_C), Write(label_C))
        
        # Show QR line
        self.play(Create(median_QR), Write(label_R))
        
        # Vector QC
        vec_QC = Arrow(Q, C, buff=0, color=RED, stroke_width=4)
        label_vec_QC = MathTex("\\vec{QC}", color=RED).next_to(vec_QC, LEFT, buff=0.1)
        
        # Vector CR (doubled logic)
        vec_CR = Arrow(C, R, buff=0, color=GREEN, stroke_width=4)
        
        self.play(GrowArrow(vec_QC), Write(label_vec_QC))
        self.wait(1)
        
        # Final Proof Text
        proof_text = MathTex(
            "\\vec{QC} = 2\\vec{CR} = \\vec{CA} + \\vec{CB}",
            color=BLUE
        ).to_edge(DOWN)
        
        self.play(Write(proof_text))
        self.wait(2)
