from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        axes = Axes(
            x_range=[-10, 5, 1],
            y_range=[-5, 15, 5],
            axis_config={"color": BLACK},
            x_length=8,
            y_length=6
        ).center()
        
        # The equation we derived: 2a^2 + 3ab + b^2 + b = 0
        # We found discrete points
        solution_points = [
            (0, 0),
            (0, -1),
            (-6, 8),
            (-6, 9)
        ]
        
        dots = VGroup(*[
            Dot(axes.c2p(a, b), color=RED, radius=0.1) 
            for a, b in solution_points
        ])
        
        # Labels for points (in English as per rules)
        labels = VGroup(*[
            MathTex(f"({a},{b})", color=BLACK, font_size=24).next_to(axes.c2p(a, b), UR, buff=0.1)
            for a, b in solution_points
        ])
        
        # Main Title (Raw text or MathTex with English only)
        title = MathTex("Integer Solutions (a, b)", color=BLACK).to_edge(UP)
        
        # Derived algebraic constraint
        constraint = MathTex("(a+3)^2 - k^2 = 8", color=BLACK).to_edge(DOWN, buff=1)

        # Rendering
        self.add(axes, dots, labels, title, constraint)
