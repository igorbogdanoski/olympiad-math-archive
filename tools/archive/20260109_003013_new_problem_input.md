from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        # Using alpha = 36 degrees
        alpha = 36 * DEGREES
        base_half = 2.5
        height = base_half * np.tan(alpha)
        
        # Points
        D = ORIGIN
        A = np.array([-base_half, 0, 0])
        B = np.array([base_half, 0, 0])
        C = np.array([0, height, 0])
        E = np.array([0, -height, 0])
        
        # Bisector calculation (A to L on BC)
        # Intersection of AL (angle 18 deg) and BC
        L = np.array([1.0, 0.5, 0]) # Approximate for diagram clarity
        
        # Geometry Objects
        tri = Polygon(A, B, C, color=BLACK, stroke_width=4)
        rhombus = Polygon(A, C, B, E, color=GRAY, stroke_width=2, stroke_opacity=0.3)
        height_line = Line(C, E, color=RED, stroke_width=3)
        bisector = Line(A, L, color=BLUE, stroke_width=4)
        
        # Labels (Strictly English/Math)
        lbl_A = MathTex("A", color=BLACK).next_to(A, LEFT)
        lbl_B = MathTex("B", color=BLACK).next_to(B, RIGHT)
        lbl_C = MathTex("C", color=BLACK).next_to(C, UP)
        lbl_D = MathTex("D", color=BLACK).next_to(D, DL, buff=0.1)
        lbl_L = MathTex("L", color=BLACK).next_to(L, UR)
        lbl_E = MathTex("E", color=BLACK).next_to(E, DOWN)
        
        # Result Text
        res = MathTex("36^\\circ, 36^\\circ, 108^\\circ", color=BLACK).to_edge(DOWN)
        
        # Adding to scene
        self.add(rhombus, tri, height_line, bisector)
        self.add(lbl_A, lbl_B, lbl_C, lbl_D, lbl_L, lbl_E, res)
