from manim import *
import numpy as np

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION ---
        # A(0,0), B(0,6), C(6,0)
        A = ORIGIN
        B = 4 * UP
        C = 4 * RIGHT
        
        triangle = Polygon(A, B, C, color=BLACK, stroke_width=4)
        
        # Points D and E on BC such that BD:DE:EC = 1:2:3
        # BC total parts = 6. B is (0,4), C is (4,0)
        vec_bc = C - B
        D = B + (1/6) * vec_bc
        E = B + (3/6) * vec_bc # Midpoint of BC for simplicity of DE=2k
        
        line_ad = Line(A, D, color=BLUE, stroke_width=3)
        line_ae = Line(A, E, color=RED, stroke_width=3)
        
        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, DL)
        lbl_B = MathTex("B", color=BLACK).next_to(B, UP)
        lbl_C = MathTex("C", color=BLACK).next_to(C, RIGHT)
        lbl_D = MathTex("D", color=BLUE).next_to(D, UR, buff=0.1)
        lbl_E = MathTex("E", color=RED).next_to(E, UR, buff=0.1)
        
        # Ratios text (English/Math only)
        ratio_text = MathTex("BD:DE:EC = 1:2:3", color=BLACK).to_edge(UP)
        
        # Angle label (DAE)
        arc = Angle(line_ae, line_ad, radius=0.7, color=BLACK)
        lbl_ang = MathTex("33.7^\\circ", color=BLACK).scale(0.7).next_to(arc, UR, buff=0.1)
        
        # Result text
        res = MathTex("Statement: FALSE", color=RED).to_edge(DOWN)
        
        self.add(triangle, line_ad, line_ae, lbl_A, lbl_B, lbl_C, lbl_D, lbl_E)
        self.add(ratio_text, arc, lbl_ang, res)
