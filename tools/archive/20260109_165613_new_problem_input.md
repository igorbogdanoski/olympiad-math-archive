from manim import *

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # Triangle ABC
        A = np.array()
        B = np.array([-4, -2, 0])
        C = np.array([4, -2, 0])
        
        triangle = Polygon(A, B, C, color=BLACK, stroke_width=4)
        
        # Point S (Intersection of cevians)
        # Using simplified barycentric for S inside
        S = np.array([0, -0.5, 0])
        
        # Intersection points on sides (D, E, F)
        D = np.array([0, -2, 0]) # S is on AD
        
        # Extend lines through S to sides
        def get_intersection(p1, p2, p3, p4):
            # Line intersection formula
            return line_intersection([p1, p2], [p3, p4])

        # Manually set points for visual balance
        E = np.array([2, 0.5, 0])
        F = np.array([-2, 0.5, 0])
        
        # Cevians
        ad = Line(A, D, color=BLACK, stroke_width=2)
        be = Line(B, E, color=BLACK, stroke_width=2)
        cf = Line(C, F, color=BLACK, stroke_width=2)
        
        # Labels
        lbl_A = MathTex("A", color=BLACK).next_to(A, UP)
        lbl_B = MathTex("B", color=BLACK).next_to(B, DL)
        lbl_C = MathTex("C", color=BLACK).next_to(C, DR)
        lbl_D = MathTex("D", color=BLACK).next_to(D, DOWN)
        lbl_E = MathTex("E", color=BLACK).next_to(E, UR)
        lbl_F = MathTex("F", color=BLACK).next_to(F, UL)
        lbl_S = MathTex("S", color=BLACK).next_to(S, UR, buff=0.1)
        
        # Distances labels (simplified)
        lbl_a = MathTex("a", color=BLUE).scale(0.8).move_to((A+S)/2).shift(LEFT*0.3)
        lbl_b = MathTex("b", color=BLUE).scale(0.8).move_to((B+S)/2).shift(UP*0.3)
        lbl_c = MathTex("c", color=BLUE).scale(0.8).move_to((C+S)/2).shift(UP*0.3)
        lbl_d = MathTex("d", color=RED).scale(0.8).move_to((S+D)/2).shift(RIGHT*0.3)
        
        # Points
        dot_s = Dot(S, color=BLACK)
        
        # Render
        self.add(triangle, ad, be, cf, dot_s)
        self.add(lbl_A, lbl_B, lbl_C, lbl_D, lbl_E, lbl_F, lbl_S)
        self.add(lbl_a, lbl_b, lbl_c, lbl_d)
        
        # Identity at bottom
        identity = MathTex("\\frac{SD}{AD} + \\frac{SE}{BE} + \\frac{SF}{CF} = 1", color=BLACK).to_edge(DOWN)
        self.add(identity)
