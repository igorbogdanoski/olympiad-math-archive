from manim import *
import numpy as np

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION (Safe 2D points) ---
        A = np.array([-1.5, 0.5, 0])
        B = np.array([1.5, 0.5, 0])
        C = np.array([0.5, 2.0, 0])
        
        # Calculate lengths
        a_len = np.linalg.norm(C - B)
        b_len = np.linalg.norm(C - A)
        c_len = np.linalg.norm(B - A)
        
        # Excenter opposite to C formula: X = (-aA - bB + cC) / (-a - b + c)
        # Note: We use the excenter relative to sides.
        X_coords = (-a_len * A - b_len * B + c_len * C) / (-a_len - b_len + c_len)
        X = X_coords
        
        # Drawing elements
        triangle = Polygon(A, B, C, color=BLACK, stroke_width=4)
        
        # Extension lines for external angles
        ext_a = Line(C, A + 1.5*(A-C), color=GRAY, stroke_style=DASHED)
        ext_b = Line(C, B + 1.5*(B-C), color=GRAY, stroke_style=DASHED)
        
        # Bisector lines
        bis_ax = Line(A, X, color=BLUE, stroke_width=3)
        bis_bx = Line(B, X, color=BLUE, stroke_width=3)
        
        # Points and Labels
        dot_x = Dot(X, color=RED)
        lbl_A = MathTex("A", color=BLACK).next_to(A, UL, buff=0.1)
        lbl_B = MathTex("B", color=BLACK).next_to(B, UR, buff=0.1)
        lbl_C = MathTex("C", color=BLACK).next_to(C, UP)
        lbl_X = MathTex("X", color=RED).next_to(X, DOWN, buff=0.1)
        
        # Angle indicator for AXB
        angle_indicator = Angle(Line(X, B), Line(X, A), radius=0.4, color=RED)
        
        # Text
        formula = MathTex("\\angle AXB = 90^\\circ - \\frac{\\gamma}{2}", color=BLACK).to_edge(DOWN)
        
        # Final render
        self.add(triangle, ext_a, ext_b, bis_ax, bis_bx, dot_x)
        self.add(lbl_A, lbl_B, lbl_C, lbl_X, angle_indicator, formula)
