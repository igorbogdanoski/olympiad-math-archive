from manim import *
import numpy as np

class SolutionScene(Scene):
    def construct(self):
        self.camera.background_color = WHITE
        
        # --- CONFIGURATION (Right triangle with A=40, B=50, C=90) ---
        C = ORIGIN
        # CA length 4, then CB length = 4 * tan(40)
        A = np.array([-4, 0, 0])
        B = np.array([0, 4 * np.tan(40 * DEGREES), 0])
        
        # Triangle
        tri = Polygon(A, B, C, color=BLACK, stroke_width=4)
        
        # Midpoint M
        M = (A + B) / 2
        
        # Foot of altitude H: projection of C onto AB
        # Line AB direction
        vec_ab = B - A
        unit_ab = vec_ab / np.linalg.norm(vec_ab)
        # Vector AC
        vec_ac = A - C
        # Projection of AC onto AB
        proj_len = np.dot(vec_ac, unit_ab)
        H = A + proj_len * unit_ab
        
        # Intersection of Angle Bisector with AB
        # Bisector slope is -1 (as CA is on -x and CB is on +y)
        # Intersection of y = -x and line through A, B
        # Line AB: y - 0 = m(x - (-4)) where m = tan(50)
        m_ab = np.tan(50 * DEGREES)
        # -x = m_ab * (x + 4) => -x = m_ab*x + 4*m_ab => x(1+m_ab) = -4*m_ab
        x_l = (-4 * m_ab) / (1 + m_ab)
        L = np.array([x_l, -x_l, 0])
        
        # Lines
        altitude = Line(C, H, color=GREY, stroke_width=3)
        median = Line(C, M, color=BLUE, stroke_width=3)
        bisector = Line(C, L, color=RED, stroke_width=3)
        
        # Labels (English only)
        lbl_A = MathTex("A", color=BLACK).next_to(A, LEFT)
        lbl_B = MathTex("B", color=BLACK).next_to(B, UP)
        lbl_C = MathTex("C", color=BLACK).next_to(C, DR)
        lbl_H = MathTex("H", color=GREY).next_to(H, UL, buff=0.1)
        lbl_M = MathTex("M", color=BLUE).next_to(M, UR, buff=0.1)
        lbl_L = MathTex("L", color=RED).next_to(L, UP, buff=0.1)
        
        # Right Angle at C
        ra = RightAngle(Line(C, A), Line(C, B), length=0.3, color=BLACK)
        
        # Final Setup
        self.add(tri, altitude, median, bisector, ra)
        self.add(lbl_A, lbl_B, lbl_C, lbl_H, lbl_M, lbl_L)
        
        # Text Result
        res = MathTex("Angle(H,C,L) = Angle(L,C,M) = 5^\\circ", color=BLACK).to_edge(DOWN)
        self.add(res)
