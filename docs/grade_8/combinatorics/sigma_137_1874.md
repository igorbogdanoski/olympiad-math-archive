    def construct(self):

        v_line = Line(main_square.get_top(), main_square.get_bottom(), color=BLACK)

        self.wait(1)

        self.play(
            Transform(label_n4, label_n7),
            Create(sub_h), 
            Create(sub_v)
        )

        self.wait(2)
